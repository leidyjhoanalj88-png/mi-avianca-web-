async function ejecutarBusqueda() { let num = document.getElementById('target').value; if(!num) return; document.getElementById('status').style.display = 'block'; document.getElementById('resultado').style.display = 'none'; try { // CUANDO SUBAS A RAILWAY, CAMBIA ESTA URL POR LA QUE TE DEN const url = `https://TU-APP-EN-RAILWAY.up.railway.app/consultar?num=${num}&tipo=3`; const response = await fetch(url); const data = await response.json(); document.getElementById('status').style.display = 'none'; let res = document.getElementById('resultado'); res.style.display = 'block'; if(data.error) { res.innerHTML = `
[!] ERROR: ${data.error.toUpperCase()}
`; } else { res.innerHTML = `
DATOS REALES - SECTOR CENTRAL
NOMBRE: ${data.nombres} ${data.apellidos}
GRUPO SISBEN: ${data.grupo}
UBICACIÓN: ${data.municipio}, ${data.departamento}
`; } } catch (e) { document.getElementById('status').innerHTML = "[!] ERROR: FALLÓ LA CONEXIÓN AL NODO"; } }
